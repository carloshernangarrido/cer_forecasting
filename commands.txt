DEPENDENCIES
- upgrade pip
    python.exe -m pip install --upgrade pip

    pip freeze > requirements.txt
    pip install -r requirements.txt

    pip install package --upgrade

    git checkout -b my_new_branch
- untrack tracked file
    git rm file_name --cached
